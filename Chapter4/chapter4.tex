\section{Framework Overview}
\label{sec:sec41}

\begin{itemize}
    \item \textbf{Goal:} Structured, practical, affordable cloud security for SMEs.
    \item \textbf{Basis:} Aligned with NIST SP 800-144, ISO/IEC 27017, ENISA.
    \item \textbf{Structure:}
    \begin{enumerate}
        \item \textbf{Governance:} Policy, Access Mgmt, Compliance.
        \item \textbf{Technical:} Network/Data controls via Open Source.
        \item \textbf{Operational:} Monitoring, Incident Response.
    \end{enumerate}
    \item \textbf{Feature:} Modular/Incremental deployment.
\end{itemize}

% ----------------------------------------------------------
\section{Framework Architecture}
\label{sec:sec42}

\subsection{Governance Layer}

\subsection{Governance Layer}
\begin{itemize}
    \item \textbf{Policy Foundation:} Roles, responsibilities, data classification.
    \item \textbf{Components:}
    \begin{itemize}
        \item Access Control (Least Privilege).
        \item Compliance Mapping (Lebanon Law 81 etc.).
        \item Awareness Training.
    \end{itemize}
\end{itemize}

\subsection{Technical Security Layer}
\begin{itemize}
    \item \textbf{Core Controls:} Open-source, scalable, low-cost.
    \item \textbf{Tools:}
    \begin{itemize}
        \item \textbf{Wazuh:} SIEM (Logs/Intrusion Detection/Compliance).
        \item \textbf{Suricata:} Network IDS/IPS.
        \item \textbf{ELK Stack:} Visualization/Dashboards.
        \item \textbf{WireGuard/OpenVPN:} Secure Access.
        \item \textbf{Rclone + GPG:} Encrypted Backups.
    \end{itemize}
\end{itemize}

\subsection{Operational Layer}
\begin{itemize}
    \item \textbf{Function:} Ongoing management and response.
    \item \textbf{Processes:}
    \begin{itemize}
        \item Continuous Log Monitoring.
        \item Vulnerability Scanning (OpenVAS).
        \item Incident Response (Procedures/Escalation).
        \item Disaster Recovery Testing.
    \end{itemize}
\end{itemize}

\section{Implementation Environment}
\label{sec:sec43}

\begin{itemize}
    \item \textbf{Setup:} Simulated SME prototype (Ubuntu/VirtualBox).
    \item \textbf{Components:}
    \begin{itemize}
        \item Host: Ubuntu 22.04 LTS.
        \item VMs: App Server, Monitor Node (Wazuh/ELK), Backup Node.
        \item Network: Suricata inline on Gateway.
    \end{itemize}
    \item \textbf{Scale:} 10-20 Users, limited bandwidth, automated configuration scripts.
\end{itemize}

\section{Simulation and Evaluation}
\label{sec:sec44}

\begin{itemize}
    \item \textbf{Test Cases:} Brute-force, Data Access, Malware.
\end{itemize}

\subsection{Evaluation Metrics}
\begin{itemize}
    \item \textbf{Accuracy:} Detection rate of simulated attacks.
    \item \textbf{Overhead:} CPU/RAM/Bandwidth usage.
    \item \textbf{Effort:} Deployment time/complexity.
    \item \textbf{Cost:} Open Source vs Commercial.
\end{itemize}

\subsection{Results Summary}
\begin{itemize}
    \item \textbf{Detection:} >90\% real-time detection.
    \item \textbf{Performance:} <15\% resource increase (acceptable).
    \item \textbf{Deployment:} <6 hours, single admin.
    \item \textbf{Cost:} <\$500 total (Hardware+Soft).
    \item \textbf{Conclusion:} Effective, Efficient, Accessible.
\end{itemize}

\section{Discussion}
\label{sec:sec45}

\begin{itemize}
    \item \textbf{Finding:} Significant security improvement possible without high commercial cost.
    \item \textbf{Adaptability:} Modular open-source tools allow incremental growth.
    \item \textbf{Alignment:} Meets International standards + Local Strategy (Lebanon).
    \item \textbf{Validation:} Expert confirmed scalability/practicality.
    \item \textbf{Challenge:} Staff training \& Maintenance updates (Future work).
\end{itemize}
